7 !====================================================================
8 ! The time_manager provides a single defined type, time_type, which is
9 ! used to store time and date quantities. A time_type is a positive
10 ! definite quantity that represents an interval of time. It can be most
11 ! easily thought of as representing the number of seconds in some time
12 ! interval. A time interval can be mapped to a date under a given calendar
13 ! definition by using it to represent the time that has passed since some
14 ! base date. A number of interfaces are provided to operate on time_type
15 ! variables and their associated calendars. Time intervals can be as large
16 ! as n days where n is the largest number represented by the default integer
17 ! type on a compiler. This is typically considerably greater than 10 million
18 ! years which is likely to be adequate for most applications. The
19 ! description of the interfaces is separated into two sections. The first
20 ! deals with operations on time intervals while the second deals with
21 ! operations that convert time intervals to dates for a given calendar.
22 !

571 
572 ! Computes date corresponding to time for gregorian calendar
573 
574 implicit none
575 
576 type(time_type), intent(in) :: time
577 integer, intent(out) :: second, minute, hour, day, month, year
578 integer :: m,t,dyear,dmonth,dday,nday,nleapyr,nfh,nhund,nfour
579 integer ndiy,nex,ibaseyr
580 logical :: leap
581 
582 ibaseyr= 1601
583 ! set nday initially to 109207 (the # of days from 1/1/1601 to 1/1/1900)
584 ! 227 years of 365 days + 72 leap years
585 nday=109207
586 ! time in seconds from base_year
587 t = time%seconds
588 nday=nday+t/(60*60*24)
589 ! find number of four hundred year periods
590 nfh=nday/146097
591 nday=modulo(nday,146097)
592 ! find number of hundred year periods
593 nhund= nday/36524
594 if(nhund.gt.3) then
595  nhund=3
596  nday=36524
597 else
598  nday=modulo(nday,36524)
599 endif
600 ! find number of four year periods
601 nfour=nday/1461
602 nday=modulo(nday,1461)
603 nex=nday/365
604 if(nex.gt.3) then
605  nex=3
606  nday=365
607 else
608  nday=modulo(nday,365)
609 endif
610 ! Is this a leap year? Gregorian calandar assigns each year evenly
611 ! divisible by 4 that is not a century year unevenly divisible by 400
612 ! as a leap-year. (i.e. 1700,1800,1900 are not leap-years, 2000 is)
613 leap=(nex.eq.3).and.((nfour.ne.24).or.(nhund.eq.3))
614  if (leap) then
615  ndiy=366
616  else
617  ndiy=365
618  endif
619 year=ibaseyr+400*nfh+100*nhund+4*nfour+nex
620 nday=nday+1
621 ! find month
622 month=0
623 do m=1,12
624  if (leap.and.(m.eq.2)) then
625  if (nday.le. (days_per_month(2)+1)) then
626  month = m
627  go to 10
628  else
629  nday = nday - (days_per_month(2)+1)
630  month = m
631  t = t - (60*60*24 * (days_per_month(2)+1))
632  endif
633  else
634  if (nday.le. days_per_month(m)) then
635  month = m
636  go to 10
637  else
638  nday = nday - days_per_month(m)
639  month = m
640  t = t - (60*60*24 * days_per_month(month))
641  endif
642  endif
643 enddo
644 10 continue
645 ! find day, hour,minute and second
646 dday = t / (60*60*24)
647 day = nday
648 t = t - dday * (60*60*24)
649 hour = t / (60 * 60)
650 t = t - hour * (60 * 60)
651 minute = t / 60
652 second = t - 60 * minute
653 !if(leap) print*,'1:t,s,m,h,d,m,y=',time,second,minute,hour,day,month,year

663 ! years being leap years.
664

674 nfour = time%days / (4 * 365 + 1)

679 if(nex == 4) then
680  nex = 3
681  day = 366
682 else
